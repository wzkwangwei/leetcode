//Reverse Integer
class Solution {
public:
    int reverse(int x) {
        int result = 0;
        int newresult = 0;
        while(x != 0)
        {
            newresult = x % 10 + result * 10;               //通过取余数依次将输入的数据反转
            if((newresult - x % 10)/10 != result)           //判断是否超过了整数的范围
            {
                return 0;
            }
            result = newresult;
            x = x/10;
        }
        return result;
    }
};
